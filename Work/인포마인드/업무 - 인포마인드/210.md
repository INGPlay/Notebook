    - 문경환 차장 문서 파기 지원

```
SELECT
					A.ACH_CD /** 업적 코드 */
					, A.JRNL_CD /** 학술지 코드 */
					, A.ISI_SE /** ISI 구분[C:ISI_SE] */
					, (SELECT AXIS.F_CS_CMN_CD('ISI_SE', A.ISI_SE) FROM DUAL) ISI_NM
					, A.JRNL_NM /** 학술지 명 */
					, A.JRNL_INST_NM /** 학술지 기관 명 */
					, A.RECD_SE /** 등재 구분[C:RECD_SE] */
					, (SELECT AXIS.F_CS_CMN_CD('RECD_SE', A.RECD_SE) FROM DUAL) RECD_NM
					, A.ISBN_CD /** ISBN 코드 */
					, A.USE_YN /** 사용 여부 */
					, A.NTN_SE /** 국가 구분[C:NTN_SE] */
					, A.IOSC_SE /** 교내외 구분[C:IOSC_SE] */
					, A.ISSN_NO /** ISSN 번호 */
					, A.IDX /** 지수 */
					, A.RMRK /** 비고 */
					, A.BFR_ACH_CD /** 이전 업적 코드 */
					, A.BFR_JRNL_CD /** 이전 학술지 코드 */
					, A.PUBMED_YN /** PUBMED 여부[D:의전원용 PubMed여부] */
          ,B.ACH_NM ---논문발표구분
				FROM DREAMY.GUJ_JRNL A
            ,DREAMY.GUJ_ACH_CD B
		 WHERE  1=1
     AND A.ACH_CD =B.ACH_CD
     AND A.USE_YN = 'Y' /** 사용 여부 */
```

		FROM DREAMY.GUJ_JRNL A ,(
			SELECT ISSN_NO,JCR_SE,IFINDEX
			FROM (
					 SELECT JCR_SE
						  , IFINDEX

						  , ACH_YR
					 FROM DREAMY.GUJ_JCR_RPTP

					 ORDER BY JCR_SE DESC ,IFINDEX DESC

		) B

		  AND A.JRNL_CD ='70486' --학술지
		  AND A.USE_YN ='Y'
		  AND A.ISSN_NO =B.ISSN_NO(+)
) A ,BGP_GRE_ARTCL_CD B
WHERE 1=1
