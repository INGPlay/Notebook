	    function fn_print(p_campSe, p_collYr, p_collSe, p_collSn, p_issuSe, p_dormrNo, p_bldgCd) {
	    	let issuUsg = $("select[name='issuUsgSe']").val();
	    	if (!issuUsg) {
	    		alert("발급용도 선택하세요.");
	    		return;
	    	}
	    	
	    	let data = {
	    		act: 'print',
	    		campSe: p_campSe,
	    		collYr: p_collYr,
	    		collSe: p_collSe,
	    		collSn: p_collSn,
	    		issuSe: p_issuSe,
	    		issuUsg: issuUsg,
	    		dormrNo: p_dormrNo,   // 아래 callReport의 idName 참조 (세션의 USER_ID로 인젝션됨). 그런데.... 아래 ajax할떄는 파라미터를 참조하여 빼니깐 오류가 나네요....
	    		bldgCd : p_bldgCd, 	  // 납입확인서 용
	    		_out: 'json'
	    	}
	
	    	$.ajax({
	    		url: '${URI}',
	    		data: data,
	    		success: function(res) {
	    			console.log(res.cnt);
	    			if (res.cnt == '1') {	    				
	    				// idName 테스트를 위하여 제거한다.
	    				// delete data.dormrNo;   
	    				
	    				## 입주확인서 출력 
	    				if (p_issuSe == "i") { 
	    					data._sn = "bWYMgXqiTwgbRFSXUKrNXBedngNITNWPfUGLACGjGijjvezynuDHnzQqQvRbvmXM"; 
	    				} 
	    				## 납입확인서 출력
	    				else if (p_issuSe == "p") {
	    					data._sn = "JGOXTOYXtsiEpqDmAINyUJvYCEGODkxTmaMfBJVObMxswGLPdCGCfqgDLKpWYBMe";   
	    				}
	    				callReport(data);
	    			}
	    		},
	    		error: function(request, status, error) { 
	    			alert("출력처리중 오류가 발생하였습니다.");
	    		}
	    	}); 
	    }
