## 개발 → 운영 이관 쿼리

<br>

- AXIS 계정으로 하기

```java
-- 메뉴 이전
INSERT INTO AXIS.CS_MENU_M@RAXIS.WORLD.COM
SELECT *
  FROM AXIS.CS_MENU_M
WHERE MENU_CD LIKE '605%'
AND MENU_LVL <>'2';

--업무권한
INSERT INTO AXIS.CS_ROLE_MENU_M@RAXIS.WORLD.COM
SELECT *
  FROM AXIS.CS_ROLE_MENU_M
WHERE MENU_CD LIKE '605%'
AND ROLE_CD = '000000';
COMMIT;
 
--업무권한H
INSERT INTO AXIS.CS_ROLE_MENU_H@RAXIS.WORLD.COM
SELECT *
  FROM AXIS.CS_ROLE_MENU_H
WHERE MENU_CD LIKE '605%'
AND ROLE_CD = '000000';
COMMIT;
 
--버튼
INSERT INTO AXIS.CS_MENU_BTN_M@RAXIS.WORLD.COM
SELECT *
  FROM AXIS.CS_MENU_BTN_M
WHERE MENU_CD LIKE '605%';
COMMIT;

--업무권한 버튼
INSERT INTO AXIS.CS_ROLE_MENU_BTN_M@RAXIS.WORLD.COM
SELECT *
  FROM AXIS.CS_ROLE_MENU_BTN_M
WHERE MENU_CD LIKE '605%'
   AND ROLE_CD = '000000';
COMMIT;
 
--업무권한 버튼H
INSERT INTO AXIS.CS_ROLE_MENU_BTN_H@RAXIS.WORLD.COM
SELECT *
  FROM AXIS.CS_ROLE_MENU_BTN_H
WHERE MENU_CD LIKE '605%'
   AND ROLE_CD = '000000';
COMMIT;
```
